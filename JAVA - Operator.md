package test;

public class ArithmetricDemo {

    public static void main(String[] args) {
        int result = 7 + 3;
        System.out.println(result);

        result = result - 1;
        System.out.println(result);

        result = result * 2;
        System.out.println(result);

        result = result / 3;
        System.out.println(result);

        result = result % 5;
        System.out.println(result);

    }

}

package test;

public class ArithmetricDemo {

    public static void main(String[] args) {
        int result = 7+3;
        System.out.println(result);

        result--;
        System.out.println(result);

        --result;
        System.out.println(result);

        result++;
        System.out.println(result);

        ++result;
        System.out.println(result);

    }

}
